  script
    async function load() {
      const username = document.getElementById('username').value;
      const res = await fetch(`apileaderboard-scoresuser=${username}`);
      const data = await res.json();

      const table = document.getElementById('results');
      table.innerHTML = `
        trthRankththBeatmapththScoreththAccuracyththModsthtr
      `;

      data.forEach(entry = {
        table.innerHTML += `
          tr
            td#${entry.rank}td
            tda href=${entry.beatmap.url} target=_blank${entry.beatmap.title}atd
            td${entry.score.toLocaleString()}td
            td${entry.accuracy}td
            td${entry.mods}td
          tr
        `;
      });
    }
  script
